public class Main
{
	public static void main(String[] args)
	{
	    Scanner sc=new Scanner(System.in);
	    int n;
	    int count=0;
	    n=sc.nextInt();
	    int a[]=new int[n];
	    for(int i=0;i<n;i++)
	    {
	        a[i]=sc.nextInt();
	    }
	    for(int i=0;i<n;i++)
	    {
	        if(a[i]<=n)
	        {
	            count++;
	        }
	    }
	    if(count==n)
	        System.out.println("Happy");
	    else
	         System.out.println("sad");
	
}
